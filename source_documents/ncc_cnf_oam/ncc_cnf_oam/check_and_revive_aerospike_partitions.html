Check and revive Aerospike partitions

# Revive dead/unknown partitions. 
kubectl -it exec <admincli> -c <container> -n <namespace> -- /opt/tpa/bin/db-roster
 
#If you see any dead on unknown partitions then revive those namespaces by using this command.
kubectl -it exec <admincli> -c <container> -n <namespace> -- /opt/aerospike/bin/asadm/asadm
 
# You will get Admin > prompt 
Admin> enable
 
#Execute manage revive for all namespaces where you see dead/unavailable partitions.
#Example
Admin+> manage revive ns dsc
~~~~~~~~~~~~~~~~~~~~Revive Namespace Partitions~~~~~~~~~~~~~~~~~~~~~
                                                       Node|Response
igaerospike-0.igaerospike-service.ig.svc.cluster.local:3010|ok    
 
#Once all of the namespaces are revived then execute recluster
 
Admin+> manage recluster
Successfully started recluster
 
#Exit the shell
 
#Run this command and make sure you don't see any dead partitions anymore.
kubectl -it exec <admincli> -c <container> -n <namespace> -- /opt/tpa/bin/db-roster
