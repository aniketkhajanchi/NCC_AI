Templates for (me/sm)-cnf-yaml files

Here are presented templates for the overrides-me.yaml and overrides-sm.yaml files.

overrides-me.yaml

global:
  istio:
    mtls:
      # Defines whether TLS is enabled on the CNF setup
      enabled: true
      # For more information check keystore configuration paragraph below
      # Absolute paths of created certificate, key and keystore files
      certificateFile: "full path of certificate file on Maestro"
      certificateKey: "full path of certificate key on Maestro"
      keystoreFile: "full path of keystore file on Maestro"
      keystorePass: "bash123"
  # Value of Ingress IP for ME site
  ingressip: "ip-address"
  # Value of Ingress port for ME site, default’s for https: 443
  ingressport: 443
  # Value for ingress protocol for ME site
  ingressproto: "https"
  diameter:
     # Value of Diameter origin host to use
     originhost: "me5"
     # Value of Diameter realm to use
     originrealm: "realm"
  # Value of SPS Site ID
  spssiteid: 5
  topologydata:
    config:
      topology:
        spssite:
        # "id" value should be same as above parameter: spssiteid.
        - id: 5
          # Value of SPS Site Name
          name: SPS-ME
  # This is the external notification server required by application pod to send 
  # the notification. This is actually our NCC's own notification server. 
  # Format is <ReleaseName used for helm install for ME site>-notification-service:8080
  externalNotificationServer: me-notification-service:8080
iohd:
  # IP for diameter calls
  diameterFloatingIP:
      - "ip-address"
  # Value of Diameter Port. Default value (for TLS):5868
  # For non-TLS, set value to 3868
  diameterPort: 5868
  # IP for 5G calls. The value should be set to ME-FQDN
  httpFloatingIP:
      - "FQDN"
  # Value of Diameter secure Port
  diameterSecurePort: 5868

overrides-sm.yaml

global:
  # Value of Ingress IP for SM site, should be set to SM-FQDN
  ingressip: "FQDN"
  # Value of Ingress Port for SM site
  ingressport: 443
  # Value of Ingress Protocol for SM site
  ingressproto: "https"
  # value of SPS Site ID
  spssiteid: 1
  topologydata:
    config:
      topology:
        spssite:
        # ”id” value should be same as above parameter: spssiteid
        - id: 1
          # Value of SPS Site Name
          name: SPS-SM
