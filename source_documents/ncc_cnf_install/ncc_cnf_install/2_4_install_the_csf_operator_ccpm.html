Install the Container Package Management operator

Create a Helm override filed named ccpm-values.yaml with these values:

helm3Only: true
controller: true
crd:
  create: false
rbac:
  enabled: false
repos:
  - "<helm_repo_name> https://customer.com/helmrepo"
 
# Update the value with service account created after installing ccpm-resource chart.
serviceAccountName: ccpm-resource-sa
namespaced:
  enabled: true
  namespaces:
    - ncc-operators
    - ig1
# Setting to false ignores certificate issues:
webhook:
  validate: false
autonomous:
  enabled: true
# Comment out tolerations and nodeSelector for NCS deployments
tolerations: []
nodeSelector: []
appHostPathMounts: []
# Helps for debugging
env:
  LOGGERS: ncms=debug
EOF

Run this command to install the chart:

helm upgrade --install app-api <helm_repo_name>/ncm-app --version 1.15.20 --namespace ncc-operators -f ccpm-values.yaml
