Start the application server

Start the application server by running this command on the recovering system.

# For SM only:
# NOTE: Get the deployment name as per your LAB:
kubectl get statefulset -n <namespace>
kubectl get deployment -n <namespace> DEPLOYMENT_NAME_TYPE=<deployment-name-type>   
# If there is common deployment name for all (iohd,centralmanagement,charging..), else you need to update deployment name in these commands
# as per your system.
kubectl scale --replicas=$CM_REPLICAS -n <namespace> statefulset $DEPLOYMENT_NAME_TYPE-centralmanagement
## On recovering system, restart the pods that were scaled down:
kubectl -n $NAMESPACE scale --replicas=1 deployment -l scaleForDbRestore=true
kubectl -n $NAMESPACE scale --replicas=$IOHD_REPLICAS statefulset $IOHD_STATEFULSET
kubectl -n $NAMESPACE scale --replicas=$LTM_REPLICAS statefulset $LTM_STATEFULSET
