Terraform inputs

Find the terraform.auto.tfvars file in the terraform files distributed by NCC.

Customize the file according to its possible inputs.

This is an example configuration:

#Example configuration.
aws-profile = "ncc"
region = "us-west-1"
permissions-boundary="nokia"
eks = {
  cluster-name = "ncc-demo"
  #Subnets identifiers from your vpc configuration in aws.
  subnets = ["subnet-08bf4459dda2db","subnet-0c611f90d5724923"]
  #security group of your account.
  security-groups = ["sg-08ba10069ae483fad"]
  #vpc identifier from your aws.
  vpc-id = "vpc-0f0ed6b526f0dd583"
  service-ipv4-cidr = "192.0.1.0/16"
  tags = {
     app = "ncc"
  }
}
# Array of 2 nodegroups as example.
nodes = [
  {
    name="aerospike-nodes"
    subnets = ["subnet-08bf4459dda2db0xfx2"]
    ami-type = "AL2_x86_64"
    instance-types = ["t3.small", "t3x.large"]
    min-size = 5
    max-size = 10
    capacity-type = "SPOT"
    max-unavailable-percentage = 50
    ec2-ssh-key = "test"
    labels = {
      aerospike = "true"
    }
    taints = [{
      key = "aerospike"
      value = "true"
      effect = "NoSchedule"
    }]
  },
  {
    name="compute-nodes"
    subnets = ["subnet-08bf4459dda2db0xfx2"]
    ami-type = "AL2_x86_64"
    instance-types = ["t3.small", "t3x.large"]
    min-size = 5
    max-size = 20
    capacity-type = "SPOT"
    max-unavailable-percentage = 50
    labels = {
      compute = "true"
    }
    taints = [{
      key = "compute"
      value = "true"
      effect = "NO_SCHEDULE"
    }]
  }
]
